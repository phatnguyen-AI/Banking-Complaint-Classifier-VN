# Training Pipeline Dependencies
# These are needed only for model training and export, NOT for production serving.
# The production API uses a separate, minimal requirements.txt in backend/.

torch               # PyTorch: model backbone and ONNX export
transformers        # Hugging Face: pre-trained model loading and Trainer API
pandas              # Data loading and manipulation
numpy               # Numerical operations (metrics computation)
scikit-learn        # Train/val/test splitting and evaluation metrics
datasets            # Hugging Face Datasets: efficient tokenization pipeline
onnx                # ONNX model format (used by convert.py for weight separation)
onnxruntime         # INT8 dynamic quantization and local inference validation
accelerate          # Required by Trainer for mixed-precision (fp16) training
jupyter             # Notebook support for EDA (01_eda_analysis.ipynb)
